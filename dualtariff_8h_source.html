27 template< uint8_t N, uint8_t OffPeakDuration = 8 >
28 class _rg_OffsetForce
29 {
30 public:
31  constexpr _rg_OffsetForce()
32  : _rg()
33  {
34  constexpr uint16_t uiPeakDurationInSec{ OffPeakDuration * 3600 };
35  // calculates offsets for force start and stop of each load
36  for (uint8_t i = 0; i != N; ++i)
37  {
38  const bool bOffsetInMinutes{ rg_ForceLoad[i].getStartOffset() > 24 || rg_ForceLoad[i].getStartOffset() < -24 };
39  const bool bDurationInMinutes{ rg_ForceLoad[i].getDuration() > 24 && UINT16_MAX != rg_ForceLoad[i].getDuration() };
40 
41  _rg[i][0] = ((rg_ForceLoad[i].getStartOffset() >= 0) ? 0 : uiPeakDurationInSec) + rg_ForceLoad[i].getStartOffset() * (bOffsetInMinutes ? 60ul : 3600ul);
42  _rg[i][0] *= 1000ul; // convert in milli-seconds
43 
44  if (UINT8_MAX == rg_ForceLoad[i].getDuration())
45  {
46  _rg[i][1] = rg_ForceLoad[i].getDuration();
47  }
48  else
49  {
50  _rg[i][1] = _rg[i][0] + rg_ForceLoad[i].getDuration() * (bDurationInMinutes ? 60ul : 3600ul) * 1000ul;
51  }
52  }
53  }
54  const uint32_t (&operator[](uint8_t i) const)[2]
55  {
56  return _rg[i];
57  }
58 
59 private:

61 };
