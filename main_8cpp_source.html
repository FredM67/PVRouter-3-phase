213 bool proceedLoadPrioritiesAndOverridingDualTariff(const int16_t currentTemperature_x100)
214 {
215  constexpr int16_t iTemperatureThreshold_x100{ iTemperatureThreshold * 100 };
216  static bool pinOffPeakState{ HIGH };
217  const auto pinNewState{ getPinState(dualTariffPin) };
218 
219  if (pinOffPeakState && !pinNewState)
220  {
221  // we start off-peak period
222  DBUGLN(F("Change to off-peak period!"));
223 
224  ul_TimeOffPeak = millis();
225 
226  if constexpr (PRIORITY_ROTATION == RotationModes::AUTO)
227  {
228  proceedRotation();
229  }
230  }
231  else
232  {
233  const auto ulElapsedTime{ static_cast< uint32_t >(millis() - ul_TimeOffPeak) };
234  const auto pinState{ getPinState(forcePin) };
235 
236  for (uint8_t i = 0; i < NO_OF_DUMPLOADS; ++i)
237  {
238  // for each load, if we're inside off-peak period and within the 'force period', trigger the ISR to turn the load ON
239  if (!pinOffPeakState && !pinNewState && (ulElapsedTime >= rg_OffsetForce[i][0]) && (ulElapsedTime < rg_OffsetForce[i][1]))
240  {
241  b_overrideLoadOn[i] = !pinState || (currentTemperature_x100 <= iTemperatureThreshold_x100);
242  }
243  else
244  {
245  b_overrideLoadOn[i] = !pinState;
246  }
247  }
248  }
249  // end of off-peak period
250  if (!pinOffPeakState && pinNewState)
251  {
252  DBUGLN(F("Change to peak period!"));
253  }
254 
255  pinOffPeakState = pinNewState;
256 
257  return (LOW == pinOffPeakState);
258 }
