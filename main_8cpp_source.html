88 ++sample_index; // increment the control flag

94 ADMUX = bit(REFS0) + sensorI[2]; // the conversion for V3 is already under way
95 ++sample_index; // increment the control flag
96 //
97 processCurrentRawSample(1, rawSample);
98 break;
99 case 4:
100 rawSample = ADC; // store the ADC value (this one is for Voltage L3)
101 ADMUX = bit(REFS0) + sensorV[0]; // the conversion for I3 is already under way
102 ++sample_index; // increment the control flag
103 //
104 processVoltageRawSample(2, rawSample);
105 break;
106 case 5:
107 rawSample = ADC; // store the ADC value (this one is for Current L3)

202bool proceedLoadPrioritiesAndOverridingDualTariff(const int16_t currentTemperature_x100)
203{
204 constexpr int16_t iTemperatureThreshold_x100{ iTemperatureThreshold * 100 };
205 static bool pinOffPeakState{ HIGH };
206 const auto pinNewState{ getPinState(dualTariffPin) };
207
208 if (pinOffPeakState && !pinNewState)
209 {
210 // we start off-peak period
211 DBUGLN(F("Change to off-peak period!"));
212
213 ul_TimeOffPeak = millis();
214
215 if constexpr (PRIORITY_ROTATION == RotationModes::AUTO)
216 {
217 proceedRotation();
218 }
219 }
220 else
221 {
222 const auto ulElapsedTime{ static_cast< uint32_t >(millis() - ul_TimeOffPeak) };
223 const auto pinState{ getPinState(forcePin) };
224
225 for (uint8_t i = 0; i < NO_OF_DUMPLOADS; ++i)
226 {
227 // for each load, if we're inside off-peak period and within the 'force period', trigger the ISR to turn the load ON
228 if (!pinOffPeakState && !pinNewState && (ulElapsedTime >= rg_OffsetForce[i][0]) && (ulElapsedTime < rg_OffsetForce[i][1]))
229 {
230 b_overrideLoadOn[i] = !pinState || (currentTemperature_x100 <= iTemperatureThreshold_x100);
231 }
232 else
233 {
234 b_overrideLoadOn[i] = !pinState;
235 }
236 }
237 }
238 // end of off-peak period
239 if (!pinOffPeakState && pinNewState)
240 {
241 DBUGLN(F("Change to peak period!"));
242 }
243
244 pinOffPeakState = pinNewState;
245
246 return (LOW == pinOffPeakState);
247}

258bool proceedLoadPrioritiesAndOverriding(const int16_t currentTemperature_x100)
259{
260 if constexpr (DUAL_TARIFF)
261 {
262 return proceedLoadPrioritiesAndOverridingDualTariff(currentTemperature_x100);
263 }
264
265 if constexpr (EMONESP_CONTROL)
266 {
267 static uint8_t pinRotationState{ HIGH };
268 const auto pinNewState{ getPinState(rotationPin) };
269
270 if (pinRotationState && !pinNewState)
271 {
272 DBUGLN(F("Trigger rotation!"));
273
274 proceedRotation();
275 }
276 pinRotationState = pinNewState;
277 }
278 else if constexpr (PRIORITY_ROTATION == RotationModes::AUTO)
279 {
280 if (ROTATION_AFTER_SECONDS < absenceOfDivertedEnergyCount)
281 {
282 proceedRotation();
283
284 absenceOfDivertedEnergyCount = 0;
285 }
286 }
287
288 if constexpr (OVERRIDE_PIN_PRESENT)
289 {
290 const auto pinState{ getPinState(forcePin) };
291
292 for (auto &bOverrideLoad : b_overrideLoadOn)
293 {
294 bOverrideLoad = !pinState;
295 }
296 }
297
298 return false;
299}
