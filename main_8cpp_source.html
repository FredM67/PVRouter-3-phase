76ISR(ADC_vect)
77{
78 static uint8_t sample_index{ 0 };
79 int16_t rawSample;
80
81 switch (sample_index)
82 {
83 case 0:
84 rawSample = ADC; // store the ADC value (this one is for Voltage L1)
85 ADMUX = bit(REFS0) + sensorV[1]; // the conversion for I1 is already under way
86 ++sample_index; // increment the control flag
87 //
88 processVoltageRawSample(0, rawSample);
89 break;
90 case 1:
91 rawSample = ADC; // store the ADC value (this one is for Current L1)
92 ADMUX = bit(REFS0) + sensorI[1]; // the conversion for V2 is already under way
93 ++sample_index; // increment the control flag
94 //
95 processCurrentRawSample(0, rawSample);
96 break;
97 case 2:
98 rawSample = ADC; // store the ADC value (this one is for Voltage L2)
99 ADMUX = bit(REFS0) + sensorV[2]; // the conversion for I2 is already under way
100 ++sample_index; // increment the control flag
101 //
102 processVoltageRawSample(1, rawSample);
103 break;
104 case 3:
105 rawSample = ADC; // store the ADC value (this one is for Current L2)
106 ADMUX = bit(REFS0) + sensorI[2]; // the conversion for V3 is already under way
107 ++sample_index; // increment the control flag
108 //
109 processCurrentRawSample(1, rawSample);
110 break;
111 case 4:
112 rawSample = ADC; // store the ADC value (this one is for Voltage L3)
113 ADMUX = bit(REFS0) + sensorV[0]; // the conversion for I3 is already under way
114 ++sample_index; // increment the control flag
115 //
116 processVoltageRawSample(2, rawSample);
117 break;
118 case 5:
119 rawSample = ADC; // store the ADC value (this one is for Current L3)
120 ADMUX = bit(REFS0) + sensorI[0]; // the conversion for V1 is already under way
121 sample_index = 0; // reset the control flag
122 //
123 processCurrentRawSample(2, rawSample);
124 break;
125 default:
126 sample_index = 0; // to prevent lockup (should never get here)
127 }
128} // end of ISR

212bool proceedLoadPrioritiesAndOverridingDualTariff(const int16_t& currentTemperature_x100)
213{
214 constexpr int16_t iTemperatureThreshold_x100{ iTemperatureThreshold * 100 };
215 static bool pinOffPeakState{ HIGH };
216 const auto pinNewState{ getPinState(dualTariffPin) };
217
218 if (pinOffPeakState && !pinNewState)
219 {
220 // we start off-peak period
221 DBUGLN(F("Change to off-peak period!"));
222
223 ul_TimeOffPeak = millis();
224
225 if constexpr (PRIORITY_ROTATION == RotationModes::AUTO)
226 {
227 proceedRotation();
228 }
229 }
230 else
231 {
232 const auto ulElapsedTime{ static_cast< uint32_t >(millis() - ul_TimeOffPeak) };
233 const auto pinState{ getPinState(forcePin) };
234
235 for (uint8_t i = 0; i < NO_OF_DUMPLOADS; ++i)
236 {
237 // for each load, if we're inside off-peak period and within the 'force period', trigger the ISR to turn the load ON
238 if (!pinOffPeakState && !pinNewState && (ulElapsedTime >= rg_OffsetForce[i][0]) && (ulElapsedTime < rg_OffsetForce[i][1]))
239 {
240 b_overrideLoadOn[i] = !pinState || (currentTemperature_x100 <= iTemperatureThreshold_x100);
241 }
242 else
243 {
244 b_overrideLoadOn[i] = !pinState;
245 }
246 }
247 }
248 // end of off-peak period
249 if (!pinOffPeakState && pinNewState)
250 {
251 DBUGLN(F("Change to peak period!"));
252 }
253
254 pinOffPeakState = pinNewState;
255
256 return (LOW == pinOffPeakState);
257}

268bool proceedLoadPrioritiesAndOverriding(const int16_t& currentTemperature_x100)
269{
270 if constexpr (DUAL_TARIFF)
271 {
272 return proceedLoadPrioritiesAndOverridingDualTariff(currentTemperature_x100);
273 }
274
275 if constexpr (EMONESP_CONTROL)
276 {
277 static uint8_t pinRotationState{ HIGH };
278 const auto pinNewState{ getPinState(rotationPin) };
279
280 if (pinRotationState && !pinNewState)
281 {
282 DBUGLN(F("Trigger rotation!"));
283
284 proceedRotation();
285 }
286 pinRotationState = pinNewState;
287 }
288 else if constexpr (PRIORITY_ROTATION == RotationModes::AUTO)
289 {
290 if (ROTATION_AFTER_SECONDS < absenceOfDivertedEnergyCount)
291 {
292 proceedRotation();
293
294 absenceOfDivertedEnergyCount = 0;
295 }
296 }
297
298 if constexpr (OVERRIDE_PIN_PRESENT)
299 {
300 const auto pinState{ getPinState(forcePin) };
301
302 for (auto &bOverrideLoad : b_overrideLoadOn)
303 {
304 bOverrideLoad = !pinState;
305 }
306 }
307
308 return false;
309}

350{

352 for (uint8_t phase = 0; phase < NO_OF_PHASES; ++phase)
353 {
354 tx_data.power_L[phase] = copyOf_sumP_atSupplyPoint[phase] / copyOf_sampleSetsDuringThisDatalogPeriod * f_powerCal[phase];
355 tx_data.power_L[phase] *= -1;
356
357 tx_data.power += tx_data.power_L[phase];
358
359 if constexpr (DATALOG_PERIOD_IN_SECONDS > 10)
360 {
361 tx_data.Vrms_L_x100[phase] = static_cast< int32_t >((100 << 2) * f_voltageCal[phase] * sqrt(copyOf_sum_Vsquared[phase] / copyOf_sampleSetsDuringThisDatalogPeriod));
362 }
363 else
364 {
365 tx_data.Vrms_L_x100[phase] = static_cast< int32_t >(100 * f_voltageCal[phase] * sqrt(copyOf_sum_Vsquared[phase] / copyOf_sampleSetsDuringThisDatalogPeriod));

370void processTemperatureData()
371{
372 uint8_t idx{ temperatureSensing.get_size() };

374 {

376

383 tx_data.temperature_x100[idx] = tmp;
384 } while (idx);

433void loop()
434{

436 static bool bOffPeak{ false };

438
439 if (b_newMainsCycle) // flag is set after every pair of ADC conversions

445 {
