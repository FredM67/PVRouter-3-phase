92void initializeProcessing()
93{
94 for (uint8_t i = 0; i < NO_OF_DUMPLOADS; ++i)
95 {
96 if (physicalLoadPin[i] < 8)
97 DDRD |= bit(physicalLoadPin[i]); // driver pin for Load #n
98 else
99 DDRB |= bit(physicalLoadPin[i] - 8); // driver pin for Load #n
100
101 loadPrioritiesAndState[i] &= loadStateMask;
102 }
103 updatePhysicalLoadStates(); // allows the logical-to-physical mapping to be changed
104
105 updatePortsStates(); // updates output pin states
106
107 for (auto &DCoffset_V : l_DCoffset_V)
108 DCoffset_V = 512L * 256L; // nominal mid-point value of ADC @ x256 scale
109
110 for (auto &bForceLoad : b_forceLoadOn)
111 bForceLoad = false;
112
113 // Set up the ADC to be free-running
114 ADCSRA = bit(ADPS0) + bit(ADPS1) + bit(ADPS2); // Set the ADC's clock to system clock / 128
115 ADCSRA |= bit(ADEN); // Enable the ADC
116
117 ADCSRA |= bit(ADATE); // set the Auto Trigger Enable bit in the ADCSRA register. Because
118 // bits ADTS0-2 have not been set (i.e. they are all zero), the
119 // ADC's trigger source is set to "free running mode".
120
121 ADCSRA |= bit(ADIE); // set the ADC interrupt enable bit. When this bit is written
122 // to one and the I-bit in SREG is set, the
123 // ADC Conversion Complete Interrupt is activated.
124
125 ADCSRA |= bit(ADSC); // start ADC manually first time
126 sei(); // Enable Global Interrupts
127}

229void processPolarity(const uint8_t phase, const int16_t rawSample)
230{
231 l_lastSampleVminusDC[phase] = l_sampleVminusDC[phase]; // required for phaseCal algorithm
232 // remove DC offset from each raw voltage sample by subtracting the accurate value
233 // as determined by its associated LP filter.
234 l_sampleVminusDC[phase] = (((int32_t)rawSample) << 8) - l_DCoffset_V[phase];
235 polarityOfMostRecentSampleV[phase] = (l_sampleVminusDC[phase] > 0) ? Polarities::POSITIVE : Polarities::NEGATIVE;

246void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
247{
248 static int32_t sampleIminusDC;
249 static int32_t phaseShiftedSampleVminusDC;
250 static int32_t filtV_div4;
251 static int32_t filtI_div4;
252 static int32_t instP;
253
254 // remove most of the DC offset from the current sample (the precise value does not matter)
255 sampleIminusDC = ((int32_t)(rawSample - l_DCoffset_I_nom)) << 8;
256 //
257 // phase-shift the voltage waveform so that it aligns with the grid current waveform
258 phaseShiftedSampleVminusDC = l_lastSampleVminusDC[phase] + (((l_sampleVminusDC[phase] - l_lastSampleVminusDC[phase]) << p_phaseCal) >> 8);
259 //
260 // calculate the "real power" in this sample pair and add to the accumulated sum
261 filtV_div4 = phaseShiftedSampleVminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
262 filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
263 instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
264 instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
265
266 l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
267 l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
268}

301void processVoltage(const uint8_t phase)
302{
303 static int32_t filtV_div4;
304 static int32_t inst_Vsquared;
305
306 // for the Vrms calculation (for datalogging only)
307 filtV_div4 = l_sampleVminusDC[phase] >> 2; // reduce to 16-bits (now x64, or 2^6)
308 inst_Vsquared = filtV_div4 * filtV_div4; // 32-bits (now x4096, or 2^12)
309 inst_Vsquared >>= 12; // scaling is now x1 (V_ADC x I_ADC)
310 l_sum_Vsquared[phase] += inst_Vsquared; // cumulative V^2 (V_ADC x I_ADC)

349 bool bOK_toAddLoad{true};
350 auto tempLoad{nextLogicalLoadToBeAdded()};
351
352 if (tempLoad >= NO_OF_DUMPLOADS)
353 return;
354
355 // a load which is now OFF has been identified for potentially being switched ON
356 if (b_recentTransition)

359 f_upperEnergyThreshold = f_energyInBucket_main;
360
361 // the energy thresholds must remain within range

363 f_upperEnergyThreshold = f_capacityOfEnergyBucket_main;

366 // wait until the recent transition has had sufficient opportunity to take effect.
367 bOK_toAddLoad = (tempLoad == activeLoad);

372 loadPrioritiesAndState[tempLoad] |= loadStateOnBit;

426void processStartNewCycle()
427{
428 // Restrictions apply for the period immediately after a load has been switched.
429 // Here the b_recentTransition flag is checked and updated as necessary.
430 // if (b_recentTransition)
431 // b_recentTransition = (++postTransitionCount < POST_TRANSITION_MAX_COUNT);
432 // for optimization, the next line is equivalent to the two lines above
433 b_recentTransition &= (++postTransitionCount < POST_TRANSITION_MAX_COUNT);
434
435 if (f_energyInBucket_main > f_midPointOfEnergyBucket_main)
436 {
437 // the energy state is in the upper half of the working range
438 f_lowerEnergyThreshold = f_lowerThreshold_default; // reset the "opposite" threshold
439 if (f_energyInBucket_main > f_upperEnergyThreshold)
440 {
441 // Because the energy level is high, some action may be required
442 proceedHighEnergyLevel();
443 }
444 }
445 else
446 {
447 // the energy state is in the lower half of the working range
448 f_upperEnergyThreshold = f_upperThreshold_default; // reset the "opposite" threshold
449 if (f_energyInBucket_main < f_lowerEnergyThreshold)
450 {
451 // Because the energy level is low, some action may be required
452 proceedLowEnergyLevel();
453 }
454 }
455
456 updatePhysicalLoadStates(); // allows the logical-to-physical mapping to be changed
457
458 updatePortsStates(); // update the control ports for each of the physical loads
459
460 // Now that the energy-related decisions have been taken, min and max limits can now
461 // be applied to the level of the energy bucket. This is to ensure correct operation
462 // when conditions change, i.e. when import changes to export, and vice versa.
463 //
464 if (f_energyInBucket_main > f_capacityOfEnergyBucket_main)
465 f_energyInBucket_main = f_capacityOfEnergyBucket_main;
466 else if (f_energyInBucket_main < 0)
467 f_energyInBucket_main = 0;
468}

477void processMinusHalfCycle(const uint8_t phase)
478{
479 // This is a convenient point to update the Low Pass Filter for removing the DC
480 // component from the phase that is being processed.
481 // The portion which is fed back into the integrator is approximately one percent
482 // of the average offset of all the SampleVs in the previous mains cycle.
483 //
484 l_DCoffset_V[phase] += (l_cumVdeltasThisCycle[phase] >> 12);
485 l_cumVdeltasThisCycle[phase] = 0;
486
487 // To ensure that this LP filter will always start up correctly when 240V AC is
488 // available, its output value needs to be prevented from drifting beyond the likely range
489 // of the voltage signal.
490 //
491 if (l_DCoffset_V[phase] < l_DCoffset_V_min)
492 l_DCoffset_V[phase] = l_DCoffset_V_min;
493 else if (l_DCoffset_V[phase] > l_DCoffset_V_max)
494 l_DCoffset_V[phase] = l_DCoffset_V_max;
495}

632void processRawSamples(const uint8_t phase)
633{
634 // The raw V and I samples are processed in "phase pairs"
635 if (Polarities::POSITIVE == polarityConfirmed[phase])
636 {
637 // the polarity of this sample is positive
638 if (Polarities::POSITIVE != polarityConfirmedOfLastSampleV[phase])
639 {
640 if (beyondStartUpPeriod)
641 {
642 // This is the start of a new +ve half cycle, for this phase, just after the zero-crossing point.
643 processPlusHalfCycle(phase);
644 }
645 else
646 processStartUp(phase);
647 }
648
649 // still processing samples where the voltage is POSITIVE ...
650 // check to see whether the trigger device can now be reliably armed
651 if (beyondStartUpPeriod && (0 == phase) && (2 == n_samplesDuringThisMainsCycle[0])) // lower value for larger sample set
652 {
653 // This code is executed once per 20mS, shortly after the start of each new mains cycle on phase 0.
654 processStartNewCycle();
655 }
656 }
657 else
658 {
659 // the polarity of this sample is negative
660 if (Polarities::NEGATIVE != polarityConfirmedOfLastSampleV[phase])
661 {
662 // This is the start of a new -ve half cycle (just after the zero-crossing point)
663 processMinusHalfCycle(phase);
664 }
665 }
666} // end of processRawSamples()
