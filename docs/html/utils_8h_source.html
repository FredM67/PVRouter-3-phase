318inline int16_t readTemperature()
319{
320 uint8_t buf[9];
321
322 if (oneWire.reset())
323 {
324 oneWire.reset();
325 oneWire.write(SKIP_ROM);
326 oneWire.write(READ_SCRATCHPAD);
327 for (auto &buf_elem : buf)
328 buf_elem = oneWire.read();
329
330 if (oneWire.crc8(buf, 8) != buf[8])
331 return BAD_TEMPERATURE;
332
333 // result is temperature x16, multiply by 6.25 to convert to temperature x100
334 int16_t result = (buf[1] << 8) | buf[0];
335 result = (result * 6) + (result >> 2);
336 if (result <= TEMP_RANGE_LOW || result >= TEMP_RANGE_HIGH)
337 return OUTOFRANGE_TEMPERATURE; // return value ('Out of range')
338
339 return result;
340 }
341 return BAD_TEMPERATURE;
342}
