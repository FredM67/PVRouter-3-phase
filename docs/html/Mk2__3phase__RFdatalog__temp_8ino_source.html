284bool proceedLoadPrioritiesAndForcing(const int16_t currentTemperature_x100)
285{
286 if constexpr (DUAL_TARIFF)
287 {
288 uint8_t i;
289 static constexpr int16_t iTemperatureThreshold_x100{iTemperatureThreshold * 100};
290 static uint8_t pinOffPeakState{HIGH};
291 const uint8_t pinNewState{!!(PIND & bit(offPeakForcePin))};
292
293 if (pinOffPeakState && !pinNewState)
294 {
295// we start off-peak period
296#ifndef NO_OUTPUT
297 Serial.println(F("Change to off-peak period!"));
298#endif // NO_OUTPUT
299 ul_TimeOffPeak = millis();
300
301 if constexpr (PRIORITY_ROTATION)
302 {
303 b_reOrderLoads = true;
304
305 // waits till the priorities have been rotated from inside the ISR
306 while (b_reOrderLoads)
307 delay(10);
308 }
309
310 // prints the (new) load priorities
311 logLoadPriorities();
312 }
313 else
314 {
315 const auto ulElapsedTime{(uint32_t)(millis() - ul_TimeOffPeak)};
316
317 for (i = 0; i < NO_OF_DUMPLOADS; ++i)
318 {
319 // for each load, if we're inside off-peak period and within the 'force period', trigger the ISR to turn the load ON
320 if (!pinOffPeakState && !pinNewState &&
321 (ulElapsedTime >= rg_OffsetForce[i][0]) && (ulElapsedTime < rg_OffsetForce[i][1]))
322 b_forceLoadOn[i] = currentTemperature_x100 <= iTemperatureThreshold_x100;
323 else
324 b_forceLoadOn[i] = false;
325 }
326 }
327// end of off-peak period
328#ifndef NO_OUTPUT
329 if (!pinOffPeakState && pinNewState)
330 Serial.println(F("Change to peak period!"));
331#endif //NO_OUTPUT
332
333 pinOffPeakState = pinNewState;
334
335 return (LOW == pinOffPeakState);
336 }
337 else
338 {
339 if constexpr (PRIORITY_ROTATION)
340 {
341 if (ROTATION_AFTER_CYCLES < absenceOfDivertedEnergyCount)
342 {
343 b_reOrderLoads = true;
344 absenceOfDivertedEnergyCount = 0;
345
346 // waits till the priorities have been rotated from inside the ISR
347 while (b_reOrderLoads)
348 delay(10);
349
350 // prints the (new) load priorities
351 logLoadPriorities();
352 }
353 }
354 return false;
355 }
356}

416void loop()
417{
418 static uint8_t perSecondTimer{0};
419 static bool bOffPeak{false};
420 static int16_t iTemperature_x100{0};
421
422 if (b_newMainsCycle) // flag is set after every pair of ADC conversions
423 {
424 b_newMainsCycle = false; // reset the flag
425 ++perSecondTimer;
426
427 if (perSecondTimer >= SUPPLY_FREQUENCY)
428 {
429 perSecondTimer = 0;
430
431 togglePin(watchDogPin);
432
433 if (!forceFullPower())
434 bOffPeak = proceedLoadPrioritiesAndForcing(iTemperature_x100); // called every second
435 }
436 }
437
438 if (b_datalogEventPending)
439 {
440 b_datalogEventPending = false;
441
442 tx_data.power = 0;
443 for (uint8_t phase = 0; phase < NO_OF_PHASES; ++phase)
444 {
445 tx_data.power_L[phase] = copyOf_sumP_atSupplyPoint[phase] / copyOf_sampleSetsDuringThisDatalogPeriod * f_powerCal[phase];
446 tx_data.power_L[phase] *= -1;
447
448 tx_data.power += tx_data.power_L[phase];
449
450 tx_data.Vrms_L_x100[phase] = (int32_t)(100 * f_voltageCal[phase] * sqrt(copyOf_sum_Vsquared[phase] / copyOf_sampleSetsDuringThisDatalogPeriod));
451 }
452
453 if constexpr (TEMP_SENSOR)
454 {
455 iTemperature_x100 = readTemperature();
456 tx_data.temperature_x100 = iTemperature_x100;
457 }
458
459 sendResults(bOffPeak);
460
461 if constexpr (TEMP_SENSOR)
462 convertTemperature(); // for use next time around
463 }
464} // end of loop()
