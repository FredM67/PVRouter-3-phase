153 //
154 // Calibration values
155 //-------------------
156 // Three calibration values are used in this sketch: f_powerCal, f_phaseCal and f_voltageCal.
157 // With most hardware, the default values are likely to work fine without
158 // need for change. A compact explanation of each of these values now follows:
159 
160 // When calculating real power, which is what this code does, the individual
161 // conversion rates for voltage and current are not of importance. It is
162 // only the conversion rate for POWER which is important. This is the
163 // product of the individual conversion rates for voltage and current. It
164 // therefore has the units of ADC-steps squared per Watt. Most systems will
165 // have a power conversion rate of around 20 (ADC-steps squared per Watt).
166 //
167 // powerCal is the RECIPR0CAL of the power conversion rate. A good value
168 // to start with is therefore 1/20 = 0.05 (Watts per ADC-step squared)
169 //
170 constexpr float f_powerCal[NO_OF_PHASES]{0.05484f, 0.05469f, 0.05385f};
171 //
172 // f_phaseCal is used to alter the phase of the voltage waveform relative to the current waveform.
173 // The algorithm interpolates between the most recent pair of voltage samples according to the value of f_phaseCal.
174 //
175 // With f_phaseCal = 1, the most recent sample is used.
176 // With f_phaseCal = 0, the previous sample is used
177 // With f_phaseCal = 0.5, the mid-point (average) value in used
178 //
179 // NB. Any tool which determines the optimal value of f_phaseCal must have a similar
180 // scheme for taking sample values as does this sketch.
181 //
182 constexpr float f_phaseCal{1};
183 //
184 // When using integer maths, calibration values that have been supplied in floating point form need to be rescaled.
185 constexpr int16_t i_phaseCal{256};
186 constexpr int16_t p_phaseCal{8};
187 //
188 // For datalogging purposes, f_voltageCal has been added too. Because the range of ADC values is
189 // similar to the actual range of volts, the optimal value for this cal factor is likely to be
190 // close to unity.
191 constexpr float f_voltageCal[NO_OF_PHASES]{0.985f, 1.003f, 0.987f} /*{1.03f, 1.03f, 1.03f}*/;

627 void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
628 {
629  static int32_t sampleIminusDC;
630  static int32_t phaseShiftedSampleVminusDC;
631  static int32_t filtV_div4;
632  static int32_t filtI_div4;
633  static int32_t instP;
634 
635  // remove most of the DC offset from the current sample (the precise value does not matter)
636  sampleIminusDC = ((int32_t)(rawSample - l_DCoffset_I_nom)) << 8;
637  //
638  // phase-shift the voltage waveform so that it aligns with the grid current waveform
639  phaseShiftedSampleVminusDC = l_lastSampleVminusDC[phase] + (((l_sampleVminusDC[phase] - l_lastSampleVminusDC[phase]) << p_phaseCal) >> 8);
640  //
641  // calculate the "real power" in this sample pair and add to the accumulated sum
642  filtV_div4 = phaseShiftedSampleVminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
643  filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
644  instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
645  instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
646 
647  l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
648  l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
649 }
