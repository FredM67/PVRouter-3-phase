543 void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
544 {
545  static int32_t sampleIminusDC;
546  static int32_t phaseShiftedSampleVminusDC;
547  static int32_t filtV_div4;
548  static int32_t filtI_div4;
549  static int32_t instP;
550 
551  // remove most of the DC offset from the current sample (the precise value does not matter)
552  sampleIminusDC = ((int32_t)(rawSample - i_DCoffset_I_nom)) << 8;
553  //
554  // phase-shift the voltage waveform so that it aligns with the grid current waveform
555  phaseShiftedSampleVminusDC = l_lastSampleVminusDC[phase] + (((l_sampleVminusDC[phase] - l_lastSampleVminusDC[phase]) * i_phaseCal) >> 8);
556  //
557  // calculate the "real power" in this sample pair and add to the accumulated sum
558  filtV_div4 = phaseShiftedSampleVminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
559  filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
560  instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
561  instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
562 
563  l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
564  l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
565 }
