374 // Three calibration values are used in this sketch: f_powerCal, f_phaseCal and f_voltageCal.
375 // With most hardware, the default values are likely to work fine without
376 // need for change. A compact explanation of each of these values now follows:
377 
378 // When calculating real power, which is what this code does, the individual
379 // conversion rates for voltage and current are not of importance. It is
380 // only the conversion rate for POWER which is important. This is the
381 // product of the individual conversion rates for voltage and current. It
382 // therefore has the units of ADC-steps squared per Watt. Most systems will
383 // have a power conversion rate of around 20 (ADC-steps squared per Watt).
384 //
385 // powerCal is the RECIPR0CAL of the power conversion rate. A good value
386 // to start with is therefore 1/20 = 0.05 (Watts per ADC-step squared)
387 //
388 constexpr float f_powerCal[NO_OF_PHASES]{0.0556f, 0.0560f, 0.0558f};
389

391 // current waveform. The algorithm interpolates between the most recent pair
392 // of voltage samples according to the value of f_phaseCal.
393 //

395 // With f_phaseCal = 0, the previous sample is used

399 // scheme for taking sample values as does this sketch.

403 // floating point form need to be rescaled.

554 void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
555 {
556  static int32_t sampleIminusDC;
557  static int32_t phaseShiftedSampleVminusDC;
558  static int32_t filtV_div4;
559  static int32_t filtI_div4;
560  static int32_t instP;
561 
562  // remove most of the DC offset from the current sample (the precise value does not matter)
563  sampleIminusDC = ((int32_t)(rawSample - i_DCoffset_I_nom)) << 8;
564  //
565  // phase-shift the voltage waveform so that it aligns with the grid current waveform
566  phaseShiftedSampleVminusDC = l_lastSampleVminusDC[phase] + (((l_sampleVminusDC[phase] - l_lastSampleVminusDC[phase]) * i_phaseCal) >> 8);
567  //
568  // calculate the "real power" in this sample pair and add to the accumulated sum
569  filtV_div4 = phaseShiftedSampleVminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
570  filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
571  instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
572  instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
573 
574  l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
575  l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
576 }

672 {
673  // The raw V and I samples are processed in "phase pairs"
674  if (Polarities::POSITIVE == polarityConfirmed[phase])
675  {
