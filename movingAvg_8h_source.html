19 /**
20  * @brief Template class for implementing a sliding average
21  *
22  * @tparam T Type of values to be stored
23  * @tparam DURATION_IN_MINUTES Size of main array
24  * @tparam VALUES_PER_MINUTE Size of sub array
25  *
26  * @note
27  * Since the Arduino RAM is tiny, we need to store a few values as possible.
28  * Suppose, you want a sliding window of 10 minutes and receive a value every 5 seconds.
29  * With one single array, 10 x 12 values would be necessary to store.
30  * Using 2 arrays, one to store incoming values over 1 minute and one to store values every minutes,
31  * we need only 12 + 10 values.
32  * How it works :
33  * - the average is calculated over 1 minute with incoming values
34  * - this sub-average is added to the main array
35  * - the average is calculated on the main array
36  *
37  * Drawback of this method: the average is updated only every minutes !
38  */
39 template< typename T, uint8_t DURATION_IN_MINUTES = 10, uint8_t VALUES_PER_MINUTE = 10 >
40 class movingAvg
41 {
42 public:
43  /**
44  * @brief Reset everything
45  *
46  */

50

54  }
55  else
56  {

58  }
59

61  do {

63  if constexpr (is_floating_point< T >::value)
64  {
65  _ar[i] = 0.0F;
66  }
67  else

70  }
71  } while (i);
72

74  }

76  /**

78  *
79  * @param _value Value to be added
80  */

85  _sub_sum += _value;

93  }

107  /**
108  * @brief Get the sliding average
109  *
110  * @return auto The sliding average
111  *
112  * @note This value is updated every minute, except for the special case of a duration of
113  * ONE minute. In this case, it is updated for each new input value.
114  */
115  auto getAverage() const
116  {
117  if constexpr (DURATION_IN_MINUTES == 1)
118  return _getAverage();
119  else if constexpr (DURATION_IN_MINUTES == 2)
120  return _sum >> 1;
121  else if constexpr (DURATION_IN_MINUTES == 4)
122  return _sum >> 2;
123  else if constexpr (DURATION_IN_MINUTES == 8)
124  return _sum >> 3;
125  else if constexpr (DURATION_IN_MINUTES == 16)
126  return _sum >> 4;
127  else if constexpr (DURATION_IN_MINUTES == 32)
128  return _sum >> 5;
129  else
130  return _sum * invN;
131  }
