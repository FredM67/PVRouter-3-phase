289 void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
290 {
291  // extra items for an LPF to improve the processing of data samples from CT1
292  static int32_t lpf_long[NO_OF_PHASES]{}; // new LPF, for offsetting the behaviour of CTx as a HPF
293 
294  // remove most of the DC offset from the current sample (the precise value does not matter)
295  int32_t sampleIminusDC = (static_cast< int32_t >(rawSample - i_DCoffset_I_nom)) << 8;
296 
297  // extra filtering to offset the HPF effect of CTx
298  const int32_t last_lpf_long{ lpf_long[phase] };
299  lpf_long[phase] = last_lpf_long + alpha * (sampleIminusDC - last_lpf_long);
300  sampleIminusDC += (lpf_gain * lpf_long[phase]);
301 
302  // calculate the "real power" in this sample pair and add to the accumulated sum
303  const int32_t filtV_div4 = l_sampleVminusDC[phase] >> 2; // reduce to 16-bits (now x64, or 2^6)
304  const int32_t filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
305  int32_t instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
306  instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
307 
308  l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
309  l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
310 }
