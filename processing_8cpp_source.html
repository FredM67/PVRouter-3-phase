300 void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
301 {
302  // extra items for an LPF to improve the processing of data samples from CT1
303  static int32_t lpf_long[NO_OF_PHASES]{}; // new LPF, for offsetting the behaviour of CTx as a HPF
304 
305  // remove most of the DC offset from the current sample (the precise value does not matter)
306  int32_t sampleIminusDC = (static_cast< int32_t >(rawSample - i_DCoffset_I_nom)) << 8;
307 
308  // extra filtering to offset the HPF effect of CTx
309  const int32_t last_lpf_long{ lpf_long[phase] };
310  lpf_long[phase] += alpha * (sampleIminusDC - last_lpf_long);
311  sampleIminusDC += (lpf_gain * lpf_long[phase]);
312 
313  // calculate the "real power" in this sample pair and add to the accumulated sum
314  const int32_t filtV_div4 = l_sampleVminusDC[phase] >> 2; // reduce to 16-bits (now x64, or 2^6)
315  const int32_t filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
316  int32_t instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
317  instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
318 
319  l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
320  l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
321 }
