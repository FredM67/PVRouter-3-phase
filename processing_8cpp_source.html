92{

94 {
95 loadPrioritiesAndState[i] = loadPrioritiesAtStartup[i];
96 pinMode(physicalLoadPin[i], OUTPUT); // driver pin for Load #n
97 loadPrioritiesAndState[i] &= loadStateMask;
98 }
99 updatePhysicalLoadStates(); // allows the logical-to-physical mapping to be changed
100
101 updatePortsStates(); // updates output pin states
102
103 for (auto &DCoffset_V : l_DCoffset_V)
104 {
105 DCoffset_V = 512L * 256L; // nominal mid-point value of ADC @ x256 scale
106 }
107
108 for (auto &bOverrideLoad : b_overrideLoadOn)
109 {
110 bOverrideLoad = false;
111 }
112
113 // First stop the ADC
114 bit_clear(ADCSRA, ADEN);
115
116 // Activate free-running mode
117 ADCSRB = 0x00;
118
119 // Set up the ADC to be free-running
120 bit_set(ADCSRA, ADPS0); // Set the ADC's clock to system clock / 128
121 bit_set(ADCSRA, ADPS1);
122 bit_set(ADCSRA, ADPS2);
123
124 bit_set(ADCSRA, ADATE); // set the Auto Trigger Enable bit in the ADCSRA register. Because
125 // bits ADTS0-2 have not been set (i.e. they are all zero), the
126 // ADC's trigger source is set to "free running mode".
127
128 bit_set(ADCSRA, ADIE); // set the ADC interrupt enable bit. When this bit is written
129 // to one and the I-bit in SREG is set, the
130 // ADC Conversion Complete Interrupt is activated.
131
132 bit_set(ADCSRA, ADEN); // Enable the ADC
133
134 bit_set(ADCSRA, ADSC); // start ADC manually first time
135
136 sei(); // Enable Global Interrupts
137}

284void processPolarity(const uint8_t phase, const int16_t rawSample)
285{
286 // remove DC offset from each raw voltage sample by subtracting the accurate value
287 // as determined by its associated LP filter.
288 l_sampleVminusDC[phase] = (static_cast< int32_t >(rawSample) << 8) - l_DCoffset_V[phase];
289 polarityOfMostRecentSampleV[phase] = (l_sampleVminusDC[phase] > 0) ? Polarities::POSITIVE : Polarities::NEGATIVE;
290}

300void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
301{
302 // extra items for an LPF to improve the processing of data samples from CT1
303 static int32_t lpf_long[NO_OF_PHASES]{}; // new LPF, for offsetting the behaviour of CTx as a HPF
304
305 // remove most of the DC offset from the current sample (the precise value does not matter)
306 int32_t sampleIminusDC = (static_cast< int32_t >(rawSample - i_DCoffset_I_nom)) << 8;
307
308 // extra filtering to offset the HPF effect of CTx
309 const int32_t last_lpf_long{ lpf_long[phase] };
310 lpf_long[phase] += alpha * (sampleIminusDC - last_lpf_long);
311 sampleIminusDC += (lpf_gain * lpf_long[phase]);
312
313 // calculate the "real power" in this sample pair and add to the accumulated sum
314 const int32_t filtV_div4 = l_sampleVminusDC[phase] >> 2; // reduce to 16-bits (now x64, or 2^6)
315 const int32_t filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
316 int32_t instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
317 instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
318
319 l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
320 l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
321}

355void processVoltage(const uint8_t phase)
356{
357 // for the Vrms calculation (for datalogging only)
358 const int32_t filtV_div4{ l_sampleVminusDC[phase] >> 2 }; // reduce to 16-bits (now x64, or 2^6)
359 int32_t inst_Vsquared{ filtV_div4 * filtV_div4 }; // 32-bits (now x4096, or 2^12)
360
361 if constexpr (DATALOG_PERIOD_IN_SECONDS > 10)
362 {
363 inst_Vsquared >>= 16; // scaling is now x1/16 (V_ADC x I_ADC)
364 }
365 else
366 {
367 inst_Vsquared >>= 12; // scaling is now x1 (V_ADC x I_ADC)
368 }
369
370 l_sum_Vsquared[phase] += inst_Vsquared; // cumulative V^2 (V_ADC x I_ADC)
371 //
372 // store items for use during next loop
373 l_cumVdeltasThisCycle[phase] += l_sampleVminusDC[phase]; // for use with LP filter
374 polarityConfirmedOfLastSampleV[phase] = polarityConfirmed[phase]; // for identification of half cycle boundaries
375 ++n_samplesDuringThisMainsCycle[phase]; // for real power calculations
376}

409void proceedHighEnergyLevel()
410{
411 bool bOK_toAddLoad{ true };
412 const auto tempLoad{ nextLogicalLoadToBeAdded() };
413
414 if (tempLoad >= NO_OF_DUMPLOADS)
415 {
416 return;
417 }
418
419 // a load which is now OFF has been identified for potentially being switched ON
420 if (b_recentTransition)
421 {
422 // During the post-transition period, any increase in the energy level is noted.
423 f_upperEnergyThreshold = f_energyInBucket_main;
424
425 // the energy thresholds must remain within range
426 if (f_upperEnergyThreshold > f_capacityOfEnergyBucket_main)
427 {
428 f_upperEnergyThreshold = f_capacityOfEnergyBucket_main;
429 }
430
431 // Only the active load may be switched during this period. All other loads must
432 // wait until the recent transition has had sufficient opportunity to take effect.
433 bOK_toAddLoad = (tempLoad == activeLoad);
434 }
435
436 if (bOK_toAddLoad)
437 {
438 loadPrioritiesAndState[tempLoad] |= loadStateOnBit;
439 activeLoad = tempLoad;
440 postTransitionCount = 0;
441 b_recentTransition = true;
442 }

496void processStartNewCycle()
497{
498 // Restrictions apply for the period immediately after a load has been switched.
499 // Here the b_recentTransition flag is checked and updated as necessary.
500 // if (b_recentTransition)
501 // b_recentTransition = (++postTransitionCount < POST_TRANSITION_MAX_COUNT);
502 // for optimization, the next line is equivalent to the two lines above
503 b_recentTransition &= (++postTransitionCount < POST_TRANSITION_MAX_COUNT);
504
505 if (f_energyInBucket_main > f_midPointOfEnergyBucket_main)
506 {
507 // the energy state is in the upper half of the working range
508 f_lowerEnergyThreshold = f_lowerThreshold_default; // reset the "opposite" threshold
509 if (f_energyInBucket_main > f_upperEnergyThreshold)
510 {
511 // Because the energy level is high, some action may be required
512 proceedHighEnergyLevel();
513 }
514 }
515 else
516 {
517 // the energy state is in the lower half of the working range
518 f_upperEnergyThreshold = f_upperThreshold_default; // reset the "opposite" threshold
519 if (f_energyInBucket_main < f_lowerEnergyThreshold)
520 {
521 // Because the energy level is low, some action may be required
522 proceedLowEnergyLevel();
523 }
524 }
525
526 updatePhysicalLoadStates(); // allows the logical-to-physical mapping to be changed
527
528 updatePortsStates(); // update the control ports for each of the physical loads
529
530 // Now that the energy-related decisions have been taken, min and max limits can now
531 // be applied to the level of the energy bucket. This is to ensure correct operation
532 // when conditions change, i.e. when import changes to export, and vice versa.
533 //
534 if (f_energyInBucket_main > f_capacityOfEnergyBucket_main)
535 {
536 f_energyInBucket_main = f_capacityOfEnergyBucket_main;
537 }
538 else if (f_energyInBucket_main < 0)
539 {
540 f_energyInBucket_main = 0;
541 }
542}

551void processMinusHalfCycle(const uint8_t phase)
552{
553 // This is a convenient point to update the Low Pass Filter for removing the DC
554 // component from the phase that is being processed.
555 // The portion which is fed back into the integrator is approximately one percent
556 // of the average offset of all the SampleVs in the previous mains cycle.
557 //
558 l_DCoffset_V[phase] += (l_cumVdeltasThisCycle[phase] >> 12);
559 l_cumVdeltasThisCycle[phase] = 0;
560
561 // To ensure that this LP filter will always start up correctly when 240V AC is
562 // available, its output value needs to be prevented from drifting beyond the likely range
563 // of the voltage signal.
564 //
565 if (l_DCoffset_V[phase] < l_DCoffset_V_min)
566 {
567 l_DCoffset_V[phase] = l_DCoffset_V_min;
568 }
569 else if (l_DCoffset_V[phase] > l_DCoffset_V_max)
570 {
571 l_DCoffset_V[phase] = l_DCoffset_V_max;
572 }
573}

733void processRawSamples(const uint8_t phase)
734{
735 // The raw V and I samples are processed in "phase pairs"
736 const auto &lastPolarity{ polarityConfirmedOfLastSampleV[phase] };
737
738 if (Polarities::POSITIVE == polarityConfirmed[phase])
739 {
740 // the polarity of this sample is positive
741 if (Polarities::POSITIVE != lastPolarity)
742 {
743 // This is the start of a new +ve half cycle, for this phase, just after the zero-crossing point.
744 if (beyondStartUpPeriod)
745 {
746 processPlusHalfCycle(phase);
747 }
748 else
749 {
750 processStartUp(phase);
751 }
752 }
753
754 // still processing samples where the voltage is POSITIVE ...
755 // check to see whether the trigger device can now be reliably armed
756 if ((0 == phase) && beyondStartUpPeriod && (2 == n_samplesDuringThisMainsCycle[0])) // lower value for larger sample set
757 {
758 // This code is executed once per 20mS, shortly after the start of each new mains cycle on phase 0.
759 processStartNewCycle();
760 }
761 }
762 else
763 {
764 // the polarity of this sample is negative
765 if (Polarities::NEGATIVE != lastPolarity)
766 {
767 // This is the start of a new -ve half cycle (just after the zero-crossing point)
768 processMinusHalfCycle(phase);
769 }
770 }
771}
