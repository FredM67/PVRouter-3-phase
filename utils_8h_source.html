168inline void printForEmonCMS(const bool bOffPeak)
169{

171
172 // Total mean power over a data logging period
173 Serial.print(F("P:"));
174 Serial.print(tx_data.power);
175
176 // Mean power for each phase over a data logging period
177 for (idx = 0; idx < NO_OF_PHASES; ++idx)
178 {
179 Serial.print(F(",P"));
180 Serial.print(idx + 1);
181 Serial.print(F(":"));
182 Serial.print(tx_data.power_L[idx]);
183 }
184 // Mean power for each load over a data logging period (in %)
185 for (idx = 0; idx < NO_OF_DUMPLOADS; ++idx)
186 {
187 Serial.print(F(",L"));
188 Serial.print(idx + 1);
189 Serial.print(F(":"));
190 Serial.print(copyOf_countLoadON[idx] * 100 * invDATALOG_PERIOD_IN_MAINS_CYCLES);
191 }
192
193 if constexpr (TEMP_SENSOR_PRESENT)
194 { // Current temperature
195 for (idx = 0; idx < temperatureSensing.get_size(); ++idx)
196 {
197 if ((OUTOFRANGE_TEMPERATURE == tx_data.temperature_x100[idx])
198 || (DEVICE_DISCONNECTED_RAW == tx_data.temperature_x100[idx]))
199 {
200 continue;
201 }
202
203 Serial.print(F(",T"));
204 Serial.print(idx + 1);
205 Serial.print(F(":"));
206 Serial.print(tx_data.temperature_x100[idx] * 0.01F);
207 }
208 }
209
210 if constexpr (DUAL_TARIFF)
211 {
212 // Current tariff
213 Serial.print(F(",T:"));
214 Serial.print(bOffPeak ? "low" : "high");
215 }
216 Serial.println(F(""));

306void sendTelemetryData()
307{
308 static TeleInfo teleInfo;
309 uint8_t idx = 0;
310
311 teleInfo.startFrame(); // Start a new telemetry frame
312
313 teleInfo.send("P", tx_data.power); // Send power grid data
314
315 if constexpr (RELAY_DIVERSION)
316 {
317 teleInfo.send("R", static_cast< int16_t >(relays.get_average())); // Send relay average if diversion is enabled
318
319 idx = 0;
320 do
321 {
322 teleInfo.send("R", relays.get_relay(idx).isRelayON()); // Send diverted energy count for each relay
323 } while (++idx < relays.get_size());
324 }
325
326 idx = 0;
327 do
328 {
329 teleInfo.send("V", tx_data.Vrms_L_x100[idx], idx + 1); // Send voltage for each phase
330 } while (++idx < NO_OF_PHASES);
331
332 idx = 0;
333 do
334 {
335 teleInfo.send("L", copyOf_countLoadON[idx] * 100 * invDATALOG_PERIOD_IN_MAINS_CYCLES, idx + 1); // Send load ON count for each load
336 } while (++idx < NO_OF_DUMPLOADS);
337
338 if constexpr (TEMP_SENSOR_PRESENT)
339 {
340 for (uint8_t idx = 0; idx < temperatureSensing.get_size(); ++idx)
341 {
342 if ((OUTOFRANGE_TEMPERATURE == tx_data.temperature_x100[idx])
343 || (DEVICE_DISCONNECTED_RAW == tx_data.temperature_x100[idx]))
344 {
345 continue; // Skip invalid temperature readings
346 }
347 teleInfo.send("T", tx_data.temperature_x100[idx], idx + 1); // Send temperature
348 }
349 }
350
351 teleInfo.send("N", static_cast< int16_t >(absenceOfDivertedEnergyCount)); // Send absence of diverted energy count for 50Hz
352
353 teleInfo.endFrame(); // Finalize and send the telemetry frame
354}
