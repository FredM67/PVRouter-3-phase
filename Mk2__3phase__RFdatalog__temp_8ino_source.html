154 //
155 // Calibration values
156 //-------------------
157 // Three calibration values are used in this sketch: f_powerCal, f_phaseCal and f_voltageCal.
158 // With most hardware, the default values are likely to work fine without
159 // need for change. A compact explanation of each of these values now follows:
160 
161 // When calculating real power, which is what this code does, the individual
162 // conversion rates for voltage and current are not of importance. It is
163 // only the conversion rate for POWER which is important. This is the
164 // product of the individual conversion rates for voltage and current. It
165 // therefore has the units of ADC-steps squared per Watt. Most systems will
166 // have a power conversion rate of around 20 (ADC-steps squared per Watt).
167 //
168 // powerCal is the RECIPR0CAL of the power conversion rate. A good value
169 // to start with is therefore 1/20 = 0.05 (Watts per ADC-step squared)
170 //
171 constexpr float f_powerCal[NO_OF_PHASES]{0.05000f, 0.05000f, 0.05000f};
172 //
173 // f_phaseCal is used to alter the phase of the voltage waveform relative to the current waveform.
174 // The algorithm interpolates between the most recent pair of voltage samples according to the value of f_phaseCal.
175 //
176 // With f_phaseCal = 1, the most recent sample is used.
177 // With f_phaseCal = 0, the previous sample is used
178 // With f_phaseCal = 0.5, the mid-point (average) value in used
179 //
180 // NB. Any tool which determines the optimal value of f_phaseCal must have a similar
181 // scheme for taking sample values as does this sketch.
182 //
183 constexpr float f_phaseCal{1};
184 //
185 // When using integer maths, calibration values that have been supplied in floating point form need to be rescaled.
186 constexpr int16_t i_phaseCal{256};
187 constexpr uint8_t p_phaseCal{8};
188 //
189 // For datalogging purposes, f_voltageCal has been added too. Because the range of ADC values is

660 void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
661 {
662  // remove most of the DC offset from the current sample (the precise value does not matter)
663  int32_t sampleIminusDC = ((int32_t)(rawSample - l_DCoffset_I_nom)) << 8;
664  //
665  // phase-shift the voltage waveform so that it aligns with the grid current waveform
666  int32_t phaseShiftedSampleVminusDC = l_lastSampleVminusDC[phase] + (((l_sampleVminusDC[phase] - l_lastSampleVminusDC[phase]) << p_phaseCal) >> 8);
667  //
668  // calculate the "real power" in this sample pair and add to the accumulated sum
669  int32_t filtV_div4 = phaseShiftedSampleVminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
670  int32_t filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
671  int32_t instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
672  instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
673 
674  l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
675  l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
676 }
