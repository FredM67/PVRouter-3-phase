222 [[nodiscard]] int16_t readTemperature(const uint8_t idx) const
223 {
224 static ScratchPad buf;
225
226 if constexpr (TEMP_SENSOR_PRESENT)
227 {
228 if (!oneWire.reset())
229 {
230 return DEVICE_DISCONNECTED_RAW;
231 }
232 oneWire.select(sensorAddrs[idx].addr);
233 oneWire.write(READ_SCRATCHPAD);
234
235 for (auto& buf_elem : buf)
236 {
237 buf_elem = oneWire.read();
238 }
239
240 if (!oneWire.reset())
241 {
242 return DEVICE_DISCONNECTED_RAW;
243 }
244 if (oneWire.crc8(buf, 8) != buf[8])
245 {
246 return DEVICE_DISCONNECTED_RAW;
247 }
248 }
249
250 // result is temperature x16, multiply by 6.25 to convert to temperature x100
251 int16_t result = (buf[1] << 8) | buf[0];
252 result = (result * 6) + (result >> 2);
253 if (result <= TEMP_RANGE_LOW || result >= TEMP_RANGE_HIGH)
254 {
255 return OUTOFRANGE_TEMPERATURE; // return value ('Out of range')
256 }
257
258 return result;
259 }
260

267};
