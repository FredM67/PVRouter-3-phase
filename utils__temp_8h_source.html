137 int16_t readTemperature(const uint8_t idx) const
138 {
139 static ScratchPad buf;
140
141#ifdef TEMP_ENABLED
142 if (!oneWire.reset())
143 {
144 return DEVICE_DISCONNECTED_RAW;
145 }
146 oneWire.select(sensorAddrs[idx].addr);
147 oneWire.write(READ_SCRATCHPAD);
148
149 for (auto &buf_elem : buf)
150 {
151 buf_elem = oneWire.read();
152 }
153
154 if (!oneWire.reset())
155 {
156 return DEVICE_DISCONNECTED_RAW;
157 }
158 if (oneWire.crc8(buf, 8) != buf[8])
159 {
160 return DEVICE_DISCONNECTED_RAW;
161 }
162#endif
163
164 // result is temperature x16, multiply by 6.25 to convert to temperature x100
165 int16_t result = (buf[1] << 8) | buf[0];
166 result = (result * 6) + (result >> 2);
167 if (result <= TEMP_RANGE_LOW || result >= TEMP_RANGE_HIGH)
168 {
169 return OUTOFRANGE_TEMPERATURE; // return value ('Out of range')
170 }
171
172 return result;
173 }
