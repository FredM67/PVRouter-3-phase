8  * @section description Description
9  * The Exponentially Weighted Moving Average (EWMA) is a quantitative or statistical measure used to model or describe a time series.
10  * The EWMA is widely used in finance, the main applications being technical analysis and volatility modeling.
11  *
12  * The moving average is designed as such that older observations are given lower weights.
13  * The weights fall exponentially as the data point gets older – hence the name exponentially weighted.
14  *
15  * The only decision a user of the EWMA must make is the parameter alpha.
16  * The parameter decides how important the current observation is in the calculation of the EWMA.
17  * The higher the value of alpha, the more closely the EWMA tracks the original time series.
18  *
19  * Computation of DEMA (Double EMA) with half-alpha has been added to get a better response of the average,
20  * especially when "peak inputs" are recorded.
21  *
22  * Computation of TEMA (Triple EMA) with quarter-alpha has been added to get a even better response of the average,
23  * especially when "peak inputs" are recorded. This seems to be the "optimal" solution.
24  *
25  * @section note Note
26  * This class is implemented in way to use only integer math.
27  * This comes with some restrictions on the alpha parameter, but the benefit of full integer math wins
28  * on the side-drawback.
29  *
30  * @copyright Copyright (c) 2024
31  *
32  */

41 /**
42  * @brief Helper compile-time function to retrieve the previous power of 2 of the given number (120 => 64 => 6)
43  *
44  * @param v The input number
45  * @return constexpr uint8_t The next power of two
46  */
47 constexpr uint8_t round_up_to_power_of_2(uint16_t v)
48 {
49  if (__builtin_popcount(v) == 1) { return __builtin_ctz(v) - 1; }
50 
51  uint8_t next_pow_of_2{ 0 };
52 
53  while (v)
54  {
55  v >>= 1;
56  ++next_pow_of_2;
57  }
58 
59  return --next_pow_of_2;
60 }

62 /**
63  * @brief Exponentially Weighted Moving Average
64  *
65  * @details The smoothing factor is the approximate amount of values taken to calculate the average.
66  * Since the Arduino is very slow and does not provide any dedicated math co-processor,
67  * the smoothing factor will be rounded to the previous power of 2. Ie 120 will be rounded to 64.
68  * This allows to perform all the calculations with integer math, which is much faster !
69  *
70  * @note Because of the 'sign extension', the sign is copied into lower bits.
71  *
72  * @tparam A Smoothing factor
73  * @param input Input value
74  * @return long Output value
75  */
76 template< uint8_t A = 10 >
77 class EWMA_average
78 {
79 public:
80  /**
81  * @brief Add a new value and actualize the EMA, DEMA and TEMA
82  *
83  * @param input The new value
84  */
85  void addValue(int32_t input)
86  {
87  ema_raw = ema_raw - ema + input;
88  ema = ema_raw >> round_up_to_power_of_2(A);
89 
90  ema_ema_raw = ema_ema_raw - ema_ema + ema;
91  ema_ema = ema_ema_raw >> (round_up_to_power_of_2(A) - 1);
92 
93  ema_ema_ema_raw = ema_ema_ema_raw - ema_ema_ema + ema_ema;
94  ema_ema_ema = ema_ema_ema_raw >> (round_up_to_power_of_2(A) - 2);
95  }
96 
97  /**
98  * @brief Get the EMA
99  *
100  * @return auto The EMA value
